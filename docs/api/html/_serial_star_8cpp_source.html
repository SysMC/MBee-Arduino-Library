473  void RxIoSampleResponse::getNextSample(uint8_t &alias)
474  {
475  switch(getData()[alias] & 0x7F) //Сбрасываем старший бит, в котором передается информация о текущем состоянии цифровой линии.
476  {
477  case IO_ADC:
478  alias += 4;
479  break;
480  case IO_DIGITAL_INPUT:
481  case IO_DIGITAL_OUTPUT_LO:
482  case IO_DIGITAL_OUTPUT_HI:
483  case IO_WAKEUP_INPUT_FALLING_EDGE:
484  case IO_WAKEUP_INPUT_RISING_EDGE:
485  alias += 2;
486  break;
487  case IO_COUNTER_INPUT1:
488  case IO_COUNTER_INPUT2:
489  alias += 6;
490  break;
491  }
492  }

785 uint8_t RemoteAtCommandRequest::getFrameData(uint8_t pos)
786 {
787  if(pos == 0)
788  {
789  return (_remoteAddress >> 8) & 0xff;
790  }
791  else if(pos == 1)
792  {
793  return _remoteAddress & 0xff;
794  }
795  else if(pos == 2)
796  {
797  return ((_acknowledge ? 0: BV(ACKNOWLEDGE_DISABLE_COMMAND_OPTIONS_BIT)) +\
798  (_applyChanges ? BV(APPLY_CHANGES_COMMAND_OPTION_BIT): 0) +\
799  (_saveChanges ? 0: BV(APPLY_CHANGES_NO_SAVE_COMMAND_OPTION_BIT)) +\
800  (_cca ? 0: BV(CCA_DISABLE_COMMAND_OPTION_BIT)) +\
801  (_encryption ? BV(ENCRYPTION_ENABLE_COMMAND_OPTION_BIT): 0) +\
802  (_sleepingDevice ? BV(SLEEP_DEVICE_COMMAND_OPTIONS_BIT): 0));
803  }

999 {

1001  resetResponse(); //Сброс предыдущих сообщений.
1002  while(available())
1003  {
1004  b = read();
1005  if(_pos > 0 && b == START_BYTE && ATAP == 2)
1006  {
1007  //Обнаружен стартовый байт до окончания обработки предыдущего пакета. Сбрасываем предыдущее сообщение и начинаем разбор заново.
1008  _response.setErrorCode(UNEXPECTED_START_BYTE);
1009  return;
1010  }
1011  if(_pos > 0 && b == ESCAPE)
1012  {
1013  if(available())
1014  {
1015  b = read();
1016  b = 0x20 ^ b; //Извлекаем символ из escape-последовательности.
1017  }
1018  else
1019  {
1020  _escape = true; //Ставим флаг того, что следующий байт будет являться вторым символом из escape-последовательности.
1021  continue;
1022  }
1023  }
1024  if(_escape == true)
1025  {
1026  b = 0x20 ^ b;
1027  _escape = false; //Предыдущий байт был escape-символом. Извлекаем символ из escape-последовательности.
1028  }
1029  if(_pos >= API_ID_INDEX) //Контрольную сумму считаем начиная с байта типа API-фрейма.
1030  _checksumTotal+= b;
1031 
1032  //Конечный автомат парсера.
1033  switch(_pos)
1034  {
1035  case 0:
1036  if(b == START_BYTE)
1037  _pos++;
1038  break;
1039  case 1:
1040  _response.setMsbLength(b);
1041  _pos++;
1042  break;
1043  case 2:
1044  _response.setLsbLength(b);
1045  _pos++;
1046  break;
1047  case 3:
1048  _response.setApiId(b);
1049  _pos++;
1050  break;
1051  default:
1052  //Далее разбираются байты, следующие за 4-ым.
1053  if(_pos > MAX_FRAME_DATA_SIZE)
1054  {
1055  _response.setErrorCode(PACKET_EXCEEDS_BYTE_ARRAY_LENGTH); //Длина пакета превышает максимально допустимую.
1056  return;
1057  }
1058  //Проверяем, не достигли ли еще конца поля данных.
1059  if(_pos == (_response.getPacketLength() + 3))
1060  {
1061  if((_checksumTotal & 0xff) == 0xff)
1062  {
1063  _response.setChecksum(b);
1064  _response.setAvailable(true);
1065  _response.setErrorCode(NO_ERROR);
1066  }
1067  else
1068  _response.setErrorCode(CHECKSUM_FAILURE); //Ошибка контрольной суммы.
1069  _response.setFrameLength(_pos - 4); //Длина поля данных не включает также поле Frame Type.
1070  _pos = 0;
1071  return;
1072  }
1073  else
1074  {

1077  }
1078  }
